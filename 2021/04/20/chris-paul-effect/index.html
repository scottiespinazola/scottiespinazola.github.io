clipper_before_during_after <- bind_rows(clippers2010, clippers2014, clippers2016, clippers2017)
winlossclippers <- clipper_before_during_after%>% 
  mutate(
    wins = case_when(
      grepl("W", Result) ~ 1, 
      grepl("L", Result) ~ 0)
)
astdef <- winlossclippers %>% group_by(Year) %>% 
  summarise(
    Wins = sum(wins), 
    TotalAST = sum(AST)
    )
fit <- lm(Wins ~ TotalAST, data = astdef)
summary(fit)
